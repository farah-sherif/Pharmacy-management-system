    private void jButton1MouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_jButton1MouseClicked
        // TODO add your handling code here:
        String Query = "select * from AGENTTBL where ANAME='" + username.getText() + "' and APASS = '" + PasswordTxt.getText() + "'";
        String Manager = "select * from Admin where Name='" + username.getText() + "' and Password = '" + PasswordTxt.getText() + "'";

        try {
            Con = DriverManager.getConnection("jdbc:sqlite:Pharmacydb.db");
            st = Con.createStatement();
            m = Con.createStatement();
            rs = st.executeQuery(Query);
            M = m.executeQuery(Manager);
            String Type = jComboBox1.getSelectedItem().toString();

            if ("Agent".equals(Type)) {
                if (rs.next()) {
                    new Medicine().setVisible(true);
                    this.dispose();
                } else {
                    JOptionPane.showMessageDialog(this, "WRONG USER NAME OR PASSWORD");
                }
                Con.close();
            }
            if ("Admin".equals(Type)) {
                if (M.next()) {
                    new Agents().setVisible(true);
                    this.dispose();
                } else {
                    JOptionPane.showMessageDialog(this, "WRONG USER NAME OR PASSWORD");
                    new Admin().setVisible(true);
                    this.dispose();
                }
                Con.close();
            }
        } catch (SQLException e) {
            System.out.println(e.getMessage());
        }


    }//GEN-LAST:event_jButton1MouseClicked

    private void timeMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_timeMouseClicked
        // TODO add your handling code here:

        try {
            Calendar cal = Calendar.getInstance();
            cal.getTime();
            SimpleDateFormat form = new SimpleDateFormat("h");
            //time.setText(form.format(cal.getTime()));
