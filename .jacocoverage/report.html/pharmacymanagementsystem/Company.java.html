    private void comscoreMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_comscoreMouseClicked
        // TODO add your handling code here:
       if (NAME.getText().equals(""))
       {
           NAME.setForeground(Color.decode("0xFF6666"));
           NAME.setText("Enter name");
       }
       else
       {
           Company_Services CS = new Company_Services();
           String st = null ;
           try {
               st = st.valueOf(CS.get_score(NAME.getText())) ;
           } catch (Exception ex) {
               JOptionPane.showMessageDialog(this, "No such company in the sysytem "); 
           }
           SCORE.setText(st);
       }
    }//GEN-LAST:event_comscoreMouseClicked
