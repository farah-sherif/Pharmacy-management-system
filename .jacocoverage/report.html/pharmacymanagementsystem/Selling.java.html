    private void Add_to_billMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_Add_to_billMouseClicked
        
        if(Medtext.getText().isEmpty() || QTY.getText().isEmpty())
        {
            JOptionPane.showMessageDialog(this, "MISSING INFORMATION");
        }
        else
        {
        float Sellingprice,price ;
        Name_Med = Medtext.getText();
        reqQty = Integer.valueOf(QTY.getText());
        price = S.getMed_Price(Name_Med);
        l++;
        Sellingprice=SP.Selling_Price_calc(Name_Med,reqQty);
        if(l==1 && Sellingprice!=0.0f){
            total+=Sellingprice;
            Billtxt.setText(Billtxt.getText()+"           ************************* PharmaCenter *************************\n"
                    +"\t ID    MEDICINE    QTY    PRICE    TOTAL\n\t "
                    +" "+l+"     "+Medtext.getText()+"         "+QTY.getText()
                    +"         "+price+"         "+total+"\n\t  ");
        }
        else if (l>1 && Sellingprice!=0.0f)
        { 
            total += Sellingprice ;
            Billtxt.setText(Billtxt.getText()+l+"     "+Medtext.getText()+"         "+QTY.getText()+"         "+price+"         "+total+"\n\t  ");
        }
        }
    }//GEN-LAST:event_Add_to_billMouseClicked
